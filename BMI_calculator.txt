weight = float(input("Enter your weight: "))
height = float(input("Enter your height: "))
bmi = weight / (height ** 2)

print(f"\n Your BMI is: {bmi:.2f}")

if bmi < 18.5:
    print("Underweight")
elif 18.5 <= bmi < 24.9:
    print("Normalweight")
elif 25 <= bmi < 29.9:
    print("Overweight")
else:
    print("Obesity")
